# ACS.202 - Force IA to join societies - Select only courtiers. Chance to scale up
character_event = {
    id = ACS.202
    hide_window = yes
    is_triggered_only = yes
    only_playable = yes

    trigger = {
        any_courtier = {
            society_member_of = pentaghast_society
        }
    }    
    
    immediate = {
        random_courtier = {
            limit = {
                ai = yes
                block_general_event_trigger = no
                is_in_society = no
                controls_religion = no
                OR = {
                    trait = schemer
                    trait = elusive_shadow
                    trait = deceitful
                    trait = ambitious
                    intrigue = 18
                }                
            }
            join_society = pentaghast_society
        }
    }
}

# ACS.204 - Sellect all members of the society
character_event = {
    id = ACS.204
    hide_window = yes    
    is_triggered_only = yes
    only_capable = yes
    prisoner = no
    is_in_society = yes

    immediate = {
        # First look for someone at court
        pentaghast_society = {
            any_society_member = {
                limit = {
                    is_incapable = no
                    prisoner = no
                    NOT = { is_inaccessible_trigger = yes }
                    NOT = { trait = celibate }                    
                    is_ill = no
                    NOT = { character = ROOT }
                    in_command = no
                    in_battle = no
                }
                character_event = { id = ACS.205 }
                clr_character_flag = had_pentaghast_ritual
            }
        }
        ROOT = { character_event = { id = ACS.207 days = 5 } }
    }
}
